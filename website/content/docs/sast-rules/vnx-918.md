## Overview

Server-Side Request Forgery occurs when an application makes an HTTP request to a destination determined by attacker-controlled input without validation. The server acts as a proxy, allowing attackers to reach internal services, cloud metadata endpoints (169.254.169.254), localhost, or arbitrary external hosts that the application server can reach but the attacker cannot directly. This maps to [CWE-918: Server-Side Request Forgery](https://cwe.mitre.org/data/definitions/918.html).

**Severity:** High | **CWE:** [CWE-918 – Server-Side Request Forgery](https://cwe.mitre.org/data/definitions/918.html)

## Why This Matters

SSRF was elevated to its own category in the OWASP Top Ten 2021 (A10) due to increasing prevalence driven by cloud adoption. In cloud environments, the EC2 metadata API at `169.254.169.254` returns IAM credentials with a single unauthenticated HTTP GET request. Attackers exploiting SSRF in AWS have harvested credentials that granted full S3 access, RDS access, and the ability to escalate to full account compromise. SSRF can also be used to port-scan internal networks, bypass authentication on internal admin APIs, and exfiltrate data through DNS lookups. The Capital One breach (2019) involved SSRF against the AWS metadata endpoint to steal IAM credentials.

## What Gets Flagged

The rule fires when HTTP request calls are made with patterns indicating user-controlled URL construction:

```python
# FLAGGED: URL comes directly from user input
import requests

def proxy_request(request):
    url = request.GET.get('url')
    return requests.get(url)  # SSRF: attacker controls destination

# FLAGGED: urllib with user input
import urllib.request
response = urllib.request.urlopen(user_supplied_url)
```

```javascript
// FLAGGED: fetch with query parameter URL
app.get('/proxy', async (req, res) => {
    const response = await fetch(req.query.url);  // SSRF
    res.send(await response.text());
});

// FLAGGED: axios with body URL
app.post('/fetch', async (req, res) => {
    const data = await axios.get(req.body.url);  // SSRF
    res.json(data.data);
});
```

```java
// FLAGGED: URL constructed from user input
String userInput = request.getParameter("url");
URL url = new URL(userInput);
URLConnection conn = url.openConnection();  // SSRF
```

```php
<?php
// FLAGGED: file_get_contents with user-controlled URL
$content = file_get_contents($_GET['url']);  // SSRF

// FLAGGED: cURL with user-controlled URL
curl_setopt($ch, CURLOPT_URL, $userInput);  // SSRF
```

```ruby
# FLAGGED: Net::HTTP with user input
require 'net/http'
Net::HTTP.get(URI(params[:url]))  # SSRF
```

```go
// FLAGGED: http.Get with user-controlled URL
func handler(w http.ResponseWriter, r *http.Request) {
    userURL := r.URL.Query().Get("url")
    resp, err := http.Get(userURL)  // SSRF
}
```

## Remediation

The primary defense is to never make server-side HTTP requests to URLs provided directly by users. Where a URL must come from user input, use strict allowlist validation:

```python
# SAFE: Allowlist of permitted domains
ALLOWED_HOSTS = {"api.example.com", "cdn.example.com"}

def safe_fetch(user_url: str):
    from urllib.parse import urlparse
    parsed = urlparse(user_url)
    if parsed.scheme not in ("http", "https"):
        raise ValueError("Only http/https allowed")
    if parsed.hostname not in ALLOWED_HOSTS:
        raise ValueError(f"Host {parsed.hostname} not in allowlist")
    # Block private IP ranges
    import ipaddress
    try:
        addr = ipaddress.ip_address(parsed.hostname)
        if addr.is_private or addr.is_loopback or addr.is_link_local:
            raise ValueError("Private/loopback addresses not allowed")
    except ValueError:
        pass  # hostname, not IP — DNS resolves later; use a DNS-safe HTTP client
    return requests.get(user_url, timeout=10)
```

```javascript
// SAFE: Use an allowlist and block internal ranges
const { URL } = require('url');
const ALLOWED_HOSTS = new Set(['api.example.com', 'cdn.example.com']);

async function safeFetch(userUrl) {
    const parsed = new URL(userUrl);
    if (!['http:', 'https:'].includes(parsed.protocol)) {
        throw new Error('Invalid protocol');
    }
    if (!ALLOWED_HOSTS.has(parsed.hostname)) {
        throw new Error(`Host ${parsed.hostname} not allowed`);
    }
    return fetch(userUrl);
}
```

```go
// SAFE: Validate URL before making request
func safeFetch(userURL string) (*http.Response, error) {
    parsed, err := url.Parse(userURL)
    if err != nil {
        return nil, err
    }
    if parsed.Scheme != "https" {
        return nil, fmt.Errorf("only https allowed")
    }
    allowed := map[string]bool{"api.example.com": true}
    if !allowed[parsed.Hostname()] {
        return nil, fmt.Errorf("host not in allowlist")
    }
    return http.Get(userURL)
}
```

### Additional Controls

1. **Block internal IP ranges** at the HTTP client level or via network egress filtering.
2. **Disable redirects** or validate the redirect destination against the same allowlist.
3. **Use a dedicated egress proxy** that enforces network-level allowlists rather than relying solely on application code.
4. **Apply a response timeout** to prevent SSRF-based port scanning via timing.
